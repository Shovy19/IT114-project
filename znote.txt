
  const [buildingData, setBuildingData] = useState(null);

  // Fetch data from Sanity
  useEffect(() => {
    client
      .fetch(
        '*[_type == "building"]{name, building_id, description, image, rooms[]->{room_id, location, description}}'
      )
      .then((data) => {
        // Set the fetched data to the state
        setBuildingData(data[0]); // Assuming you want data for the first building, you can adjust as needed
      })
      .catch((error) => {
        console.error('Error fetching data from Sanity:', error);
      });
  }, []); // The empty array ensures the effect runs only once when the component mounts

  if (!buildingData) {
    return <div>Loading...</div>;
  }


<section className='pt-40 pb-5 px-3 grid lg:grid-cols-4 gap-3 text-center'>
        <article className='bg-slate-500 p-1 rounded-lg shadow-lg'>
          <img src={buildingData.image.asset.url} alt={buildingData.name} />  
          <h1 className='py-2 text-slate-100 font-semibold'>{buildingData.name}</h1>
          <p>{buildingData.description}</p>
        </article>
